\begin{minted}{make}
    LANGUAGE=python3
    TXT_FILES=$(wildcard books/*.txt)
    DAT_FILES=$(patsubst books/%.txt, %.dat, $(TXT_FILES))
    
    results.txt : testzipf.py $(DAT_FILES)
        $(LANGUAGE) $^ > $@
    
    .PHONY : dats
    dats : $(DAT_FILES)
        
    %.dat : countwords.py books/%.txt
        $(LANGUAGE) $^ $@
    
    .PHONY : clean
    clean :
        rm -f $(DAT_FILES) results.txt
    
    .PHONY : variables
    variables:
        @echo TXT_FILES: $(TXT_FILES)
        @echo DAT_FILES: $(DAT_FILES)
    \end{minted}
