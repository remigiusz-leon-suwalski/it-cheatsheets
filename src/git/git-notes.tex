\section{To be done}
To remove a commit:
\begin{minted}{bash}
git rebase --onto ~<sha>
\end{minted}

To remove a submodule:
\begin{minted}{bash}
git submodule deinit <name>
git rm <name>
git rm --cached <name>
rm -rf .git/modules/<name>
\end{minted}

To remove a tag:
\begin{minted}{bash}
git tag -d <tag>
git push <remote> :refs/tags/<tag>
\end{minted}

To undo a change:
\begin{minted}{bash}
git reset HEAD -- # unstage
git reset --mixed HEAD~ # uncommit
git checkout -- # discard
\end{minted}
